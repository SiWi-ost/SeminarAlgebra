\begin{tikzpicture}[>=latex,thick,scale=\skala,
declare function={
	sqr(\t) = \t * \t;
	Rplus(\t,\e)  = \a * sqrt(cos(2*\t) + sqrt(\e*\e*\e*\e-sqr(sin(2*\t))));
	Rminus(\t,\e) = \a * sqrt(cos(2*\t) - sqrt(\e*\e*\e*\e-sqr(sin(2*\t))));
}]
