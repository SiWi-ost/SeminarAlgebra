\documentclass[tikz]{standalone}
\usepackage{amsmath}
\usepackage{times}
\usepackage{txfonts}
\usepackage{pgfplots}
\usepackage{csvsimple}
\usetikzlibrary{arrows,intersections,math}
\definecolor{darkred}{rgb}{0.8,0,0}
\begin{document}
\def\skala{1}
\def\R{2.5}
\def\r{1.2}
\def\w{56}
\begin{tikzpicture}[>=latex,thick,scale=\skala]

\begin{scope}[xshift=-3.2cm]
	\fill[color=gray!20] (0,0) -- (0:\r) arc (0:\w:\r) -- cycle;
	\node[color=blue] at ({0.5*\w}:{0.7*\r}) {$x$};
	\draw[color=gray!50] (0,0) circle[radius=\R];

	\draw[->] (-2.7,0) -- (3,0) coordinate[label={$\operatorname{Re}$}];

	\draw[color=darkred,line width=1.2pt]
		(0,0)
		-- ({\R*cos(\w)},0)
		-- (\w:\R);
	\draw[color=blue,line width=1.2pt] (0,0) -- (\w:\R);
	\fill[color=blue] (\w:\R) circle[radius=0.05];
	\node[color=blue] at (\w:\R) [above right] {$e^{ix}$};

	\node[color=darkred] at ({0.5*\R*cos(\w)},0) [below] {$y$};
	\node[color=darkred] at ({\R*cos(\w)},{0.5*\R*sin(\w)})
		[below,rotate=90] {$i\!\sqrt{1-y^2}$};

	\draw[->] (0,-2.7) -- (0,3)
		coordinate[label={right:$\operatorname{Im}$}];

	\fill[color=darkred] (0,0) circle[radius=0.05];
\end{scope}

\begin{scope}[xshift=3.2cm]
	\fill[color=gray!20] (0,0) -- (0:\r) arc (0:{90-\w}:\r) -- cycle;
	\node[color=blue] at ({0.5*(90-\w)}:{0.7*\r}) {$x$};

	\draw[color=gray!50] (0,0) circle[radius=\R];

	\draw[->] (-2.7,0) -- (3,0) coordinate[label={$\operatorname{Re}$}];

	\draw[color=darkred,line width=1.2pt]
		(0,0)
		-- ({\R*sin(\w)},0)
		-- ({90-\w}:\R);

	\draw[color=blue,line width=1.2pt] (0,0) -- ({90-\w}:\R);
	\fill[color=blue] ({90-\w}:\R) circle[radius=0.05];
	\node[color=blue] at ({90-\w}:\R) [above right] {$e^{ix}$};

	\node[color=darkred] at ({\R*sin(\w)},{0.5*\R*cos(\w)}) [right] {$iy$};
	\node[color=darkred] at ({0.5*\R*sin(\w)},0)
		[below] {$\!\sqrt{1-y^2}$};

	\draw[->] (0,-2.7) -- (0,3)
		coordinate[label={right:$\operatorname{Im}$}];

	\fill[color=darkred] (0,0) circle[radius=0.05];
\end{scope}

\end{tikzpicture}
\end{document}
